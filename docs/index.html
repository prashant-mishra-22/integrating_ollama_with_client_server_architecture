Code Implementation

# Server Implementation (s.py)
def handle_client(conn, addr):
    """Handle client connection with authentication"""
    print(f"[NEW CONNECTION] {addr} connected.")
    authenticated = authenticate(conn)
    
    if authenticated:
        while True:
            msg = receive_message(conn)
            if not msg or msg == DISCONNECT_MESSAGE:
                break
            # Filter cuss words
            if contains_cuss_words(msg):
                send_warning(conn)
                break
            # Generate AI response
            response = create_response(msg)
            send_response(conn, response)
    
    conn.close()

Installation Steps

# Clone the repository
git clone https://github.com/prashant-mishra-22/integrating_ollama_with_client_server_architecture.git

# Install dependencies
pip install ollama

# Start Ollama service
ollama serve

# Pull LLaMA3.2 model
ollama pull llama3.2

# Run the server
python s.py

# Run the client (in another terminal)
python c.py
